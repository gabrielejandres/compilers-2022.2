S -> A ; S
    | print E
    | ε
A -> id { Print( lexema ); } = E { Print( "="); }
E -> T E'
E' -> + T { Print( "+"); } E'
    | - T { Print( "-"); } E'
    | ε
T -> F T'
T' -> * F { Print( "*"); } T'
    | / F { Print( "/"); } T'
    | ε
F -> id { Print(lexema + "@"); }
   | num { Print( lexema ); }
   | ( E )
